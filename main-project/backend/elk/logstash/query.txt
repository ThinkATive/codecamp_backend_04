SELECT Product.id AS Product_id,
       Product.productName AS Product_productName,
       Product.productPrice AS Product_productPrice, 
       Product.productDescription AS Product_productDescription, 
       Product.productSerialNumber AS Product_productSerialNumber, 
       Product.productDiscount AS Product_productDiscount, 
       Product.productPeopleLiked AS Product_productPeopleLiked, 
       Product.productManufactureDate AS Product_productManufactureDate, 
       Product.updatedAt AS Product_updatedAt, 
       Product.isStock AS Product_isStock, 
       Product.productPeopleInterested AS Product_productPeopleInterested, 
       Product.deletedAt AS Product_deletedAt, 
       Product.brandId AS Product_brandId, 
       Product.seasonId AS Product_seasonId, 
       Product.productSubCategoryId AS Product_productSubCategoryId, 
       Product__Product_productSubCategory.id AS Product__Product_productSubCategory_id, 
       Product__Product_productSubCategory.productSubCategoryName AS Product__Product_productSubCategory_productSubCategoryName, 
       Product__Product_productSubCategory.productMainCategoryId AS Product__Product_productSubCategory_productMainCategoryId, 
       Product__Product_brand.id AS Product__Product_brand_id, 
       Product__Product_brand.brandName AS Product__Product_brand_brandName, 
       Product__Product_brand.deletedAt AS Product__Product_brand_deletedAt, 
       Product__Product_season.id AS Product__Product_season_id, 
       Product__Product_season.seasonName AS Product__Product_season_seasonName, 
       Product__Product_season.deletedAt AS Product__Product_season_deletedAt, 
       Product__Product_materials.id AS Product__Product_materials_id, 
       Product__Product_materials.materialName AS Product__Product_materials_materialName, 
       Product__Product_materials.deletedAt AS Product__Product_materials_deletedAt, 
       Product__Product_colors.id AS Product__Product_colors_id, 
       Product__Product_colors.colorName AS Product__Product_colors_colorName, 
       Product__Product_colors.deletedAt AS Product__Product_colors_deletedAt, 
       Product__Product_sizes.id AS Product__Product_sizes_id, 
       Product__Product_sizes.sizeName AS Product__Product_sizes_sizeName, 
       Product__Product_sizes.deletedAt AS Product__Product_sizes_deletedAt, 
       Product__Product_genders.id AS Product__Product_genders_id, 
       Product__Product_genders.genderName AS Product__Product_genders_genderName, 
       Product__Product_genders.deletedAt AS Product__Product_genders_deletedAt 
  FROM product Product
LEFT JOIN product_sub_category Product__Product_productSubCategory 
    ON Product__Product_productSubCategory.id=Product.productSubCategoryId
LEFT JOIN brand Product__Product_brand 
    ON Product__Product_brand.id=Product.brandId 
   AND (Product__Product_brand.deletedAt IS NULL)
LEFT JOIN season Product__Product_season 
    ON Product__Product_season.id=Product.seasonId 
   AND (Product__Product_season.deletedAt IS NULL)  
LEFT JOIN product_materials_material Product_Product__Product_materials
    ON Product_Product__Product_materials.productId=Product.id 
LEFT JOIN material Product__Product_materials 
    ON Product__Product_materials.id=Product_Product__Product_materials.materialId 
   AND (Product__Product_materials.deletedAt IS NULL)  
LEFT JOIN product_colors_color Product_Product__Product_colors 
    ON Product_Product__Product_colors.productId=Product.id 
LEFT JOIN color Product__Product_colors 
    ON Product__Product_colors.id=Product_Product__Product_colors.colorId 
   AND (Product__Product_colors.deletedAt IS NULL)  
LEFT JOIN product_sizes_size Product_Product__Product_sizes 
    ON Product_Product__Product_sizes.productId=Product.id 
LEFT JOIN size Product__Product_sizes 
    ON Product__Product_sizes.id=Product_Product__Product_sizes.sizeId 
   AND (Product__Product_sizes.deletedAt IS NULL)  
LEFT JOIN product_genders_gender Product_Product__Product_genders 
    ON Product_Product__Product_genders.productId=Product.id
LEFT JOIN gender Product__Product_genders 
    ON Product__Product_genders.id=Product_Product__Product_genders.genderId 
   AND (Product__Product_genders.deletedAt IS NULL) 
 WHERE Product.deletedAt IS NULL
